129
130 while ((pos = message.find("\033[", start)) != std::string::npos) {
131 // Output the part of the string before the ANSI code
132 if (pos > start) {
133 std::cout << message.substr(start, pos - start);
134 }
135
136 // Find the end of the ANSI code
