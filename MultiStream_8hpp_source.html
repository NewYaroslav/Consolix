114 std::string::size_type start = 0;
115 std::string::size_type pos = 0;
116
117 HANDLE handle_stdout = GetStdHandle(STD_OUTPUT_HANDLE);
118
119 while ((pos = message.find("\033[", start)) != std::string::npos) {
120 // Output the part of the string before the ANSI code
121 if (pos > start) {
122 std::cout << message.substr(start, pos - start);
123 }
124
125 // Find the end of the ANSI code
126 std::string::size_type end_pos = message.find('m', pos);
127 if (end_pos != std::string::npos) {
128 // Extract the ANSI code
129 std::string ansi_code = message.substr(pos + 2, end_pos - pos - 2);
130 apply_color_from_ansi_code(ansi_code, handle_stdout);
131
132 // Update position
133 start = end_pos + 1;
134 } else {
135 break;
136 }
137 }
138
139 // Output any remaining part of the message
140 if (start < message.size()) {
141 std::cout << message.substr(start);
142 }
143 if (!message.empty()) std::cout << std::endl;
144
145 // Reset the console color to default
146 SetConsoleTextAttribute(handle_stdout, to_windows_color(CONSOLIX_DEFAULT_COLOR));
147 }
