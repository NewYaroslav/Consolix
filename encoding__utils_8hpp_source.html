64 bool is_valid_utf8(const char* message) {
65 if (!message) return true;
66 const unsigned char* bytes = reinterpret_cast<const unsigned char*>(message);
67 while (*bytes != 0x00) {
68 int num = 0;
69 unsigned int cp = 0;
70 if ((*bytes & 0x80) == 0x00) { cp = (*bytes & 0x7F); num = 1; }
71 else if ((*bytes & 0xE0) == 0xC0) { cp = (*bytes & 0x1F); num = 2; }
72 else if ((*bytes & 0xF0) == 0xE0) { cp = (*bytes & 0x0F); num = 3; }
73 else if ((*bytes & 0xF8) == 0xF0) { cp = (*bytes & 0x07); num = 4; }
74 else return false;
75
76 bytes++;
77 for (int i = 1; i < num; ++i) {
78 if ((*bytes & 0xC0) != 0x80) return false;
79 cp = (cp << 6) | (*bytes & 0x3F);
80 bytes++;
81 }
82
83 if (cp > 0x10FFFF || (cp >= 0xD800 && cp <= 0xDFFF)) return false;
84 }
85 return true;
86 }
